Google Cloud Console이 이 브라우저 환경에서 로딩이 안 됩니다. 대신 **PowerShell에서 직접 배포하는 방법**을 안내드릴게요. 순서대로 따라오시면 됩니다.

---

## 1단계 — 파일 2개 만들기

PowerShell에서 아래 명령어를 실행해서 폴더와 파일을 생성합니다:

```powershell
mkdir C:\pipedrive-webhook
cd C:\pipedrive-webhook
```

그 다음 `index.js` 파일 생성:

```powershell
@'
const { google } = require("googleapis");

const SPREADSHEET_ID = "1wsxe8PybUmfD-Mz4NX28zND3a5ekmr7dK6f1amUfiAI";
const SHEET_NAME = "deals";
const TARGET_PIPELINES = [13, 10000];
const FIELD_CUSTOMER_DIVISION = "e70d9a0bad76099430ce08093516dee63085b881";
const FIELD_CUSTOMER_TYPE = "824db0958f2bb9678c8f09d7f6b8041be379e59d";
const HEADERS = ["수신일시","이벤트","딜ID","딜제목","상태","스테이지ID","조직명","소유자","통화","고객구분","고객유형","상품명","결제시작일","단가","수량","금액"];

async function getSheet() {
  const auth = new google.auth.GoogleAuth({ scopes: ["https://www.googleapis.com/auth/spreadsheets"] });
  const sheets = google.sheets({ version: "v4", auth });
  // 시트 목록 확인
  const meta = await sheets.spreadsheets.get({ spreadsheetId: SPREADSHEET_ID });
  const sheetExists = meta.data.sheets.some(s => s.properties.title === SHEET_NAME);
  if (!sheetExists) {
    await sheets.spreadsheets.batchUpdate({
      spreadsheetId: SPREADSHEET_ID,
      requestBody: { requests: [{ addSheet: { properties: { title: SHEET_NAME } } }] }
    });
    await sheets.spreadsheets.values.update({
      spreadsheetId: SPREADSHEET_ID,
      range: `${SHEET_NAME}!A1`,
      valueInputOption: "RAW",
      requestBody: { values: [HEADERS] }
    });
  }
  return sheets;
}

async function findRowsByDealId(sheets, dealId) {
  const res = await sheets.spreadsheets.values.get({ spreadsheetId: SPREADSHEET_ID, range: `${SHEET_NAME}!A:C` });
  const rows = res.data.values || [];
  const result = [];
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][2]) === String(dealId)) result.push(i + 1);
  }
  return result;
}

exports.pipedrive = async (req, res) => {
  res.set("Access-Control-Allow-Origin", "*");
  if (req.method !== "POST") return res.status(200).send("Pipedrive webhook receiver OK");
  try {
    const payload = req.body;
    const action = payload.meta?.action;
    const entity = payload.meta?.entity;
    const data = payload.data || {};

    if (entity !== "deal") return res.status(200).json({ status: "ignored", message: "Not a deal event" });

    const pipelineId = data.pipeline_id || null;
    if (pipelineId && !TARGET_PIPELINES.includes(pipelineId)) {
      return res.status(200).json({ status: "ignored", message: `Pipeline ${pipelineId} not in target list` });
    }

    const sheets = await getSheet();
    const now = new Date().toISOString();
    const dealId = data.id || "";
    const title = data.title || "";
    const status = data.status || "";
    const stageId = data.stage_id || "";
    const orgName = data.org_name || (data.org_id && data.org_id.name) || "";
    const owner = (data.user_id && data.user_id.name) || data.owner_name || "";
    const currency = data.currency || "";
    const custDiv = data[FIELD_CUSTOMER_DIVISION] || "";
    const custType = data[FIELD_CUSTOMER_TYPE] || "";
    const products = data.products_attached || [];

    const makeRow = (p) => p
      ? [now, action, dealId, title, status, stageId, orgName, owner, currency, custDiv, custType, p.name||"", p.start_date||"", p.item_price||"", p.quantity||"", p.sum||""]
      : [now, action, dealId, title, status, stageId, orgName, owner, currency, custDiv, custType, "", "", "", "", ""];

    if (action === "create") {
      const rows = products.length === 0 ? [makeRow(null)] : products.map(makeRow);
      await sheets.spreadsheets.values.append({
        spreadsheetId: SPREADSHEET_ID, range: `${SHEET_NAME}!A1`,
        valueInputOption: "RAW", requestBody: { values: rows }
      });
    } else if (action === "update") {
      const existingRows = await findRowsByDealId(sheets, dealId);
      if (existingRows.length === 0) return res.status(200).json({ status: "ignored", message: "Deal not found for update" });
      // 기존 행 역순 삭제
      for (let i = existingRows.length - 1; i >= 0; i--) {
        await sheets.spreadsheets.batchUpdate({
          spreadsheetId: SPREADSHEET_ID,
          requestBody: { requests: [{ deleteDimension: { range: { sheetId: 0, dimension: "ROWS", startIndex: existingRows[i]-1, endIndex: existingRows[i] } } }] }
        });
      }
      const rows = products.length === 0 ? [makeRow(null)] : products.map(makeRow);
      await sheets.spreadsheets.values.append({
        spreadsheetId: SPREADSHEET_ID, range: `${SHEET_NAME}!A1`,
        valueInputOption: "RAW", requestBody: { values: rows }
      });
    } else if (action === "delete") {
      const existingRows = await findRowsByDealId(sheets, dealId);
      for (let i = existingRows.length - 1; i >= 0; i--) {
        await sheets.spreadsheets.batchUpdate({
          spreadsheetId: SPREADSHEET_ID,
          requestBody: { requests: [{ deleteDimension: { range: { sheetId: 0, dimension: "ROWS", startIndex: existingRows[i]-1, endIndex: existingRows[i] } } }] }
        });
      }
    }

    return res.status(200).json({ status: "ok", action, dealId });
  } catch (err) {
    console.error(err);
    return res.status(200).json({ status: "error", message: err.message });
  }
};
'@ | Out-File -FilePath "C:\pipedrive-webhook\index.js" -Encoding utf8
```

  }
}
'@ | Out-File -FilePath "C:\pipedrive-webhook\package.json" -Encoding utf8

## 3단계 — 배포

gcloud가 있으면 아래 명령어로 한 번에 배포됩니다:

```powershell
cd C:\pipedrive-webhook
gcloud functions deploy pipedrive-webhook --gen2 --runtime nodejs22 --trigger-http --allow-unauthenticated --entry-point pipedrive --region asia-northeast3
```
